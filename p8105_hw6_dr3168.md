the data

``` r
bootstrap_results = 
  boot_straps %>% 
  mutate(
    models = map(strap_sample, 
                 ~lm(tmax ~ tmin, data = .x) ),
    results_tidy= map(models, broom::tidy),
    results_glance= map(models, broom::glance)) %>% 
  select(-strap_sample, -models) %>% 
  unnest(results_glance) %>% 
  unnest(results_tidy, names_repair = "unique")
```
